38// Elias Gamma - The fundamental universal code
39// ============================================================
40
41struct EliasGamma {
42 template<std::unsigned_integral T, typename Sink>
43 static void encode(T value, Sink& sink) requires BitSink<Sink> {
44 ++value; // Map 0→1, 1→2, etc. for gamma coding
45
46 int len = msb_position(value) + 1;
47
48 // Write len-1 zeros
49 for (int i = 0; i < len - 1; ++i) {
50 sink.write(false);
51 }
52
53 // Write value in binary (MSB to LSB)
54 for (int i = len - 1; i >= 0; --i) {
55 sink.write((value >> i) & 1);
56 }
57 }
58
59 template<std::unsigned_integral T>
60 static T decode(auto& source) requires BitSource<decltype(source)> {
61 // Count leading zeros
62 int zeros = 0;
63 while (!source.read()) {
64 ++zeros;
65 }
66
67 // Read zeros more bits
68 T value = 1;
69 for (int i = 0; i < zeros; ++i) {
70 value = (value << 1) | source.read();
71 }
72
73 return value - 1; // Map back: 1→0, 2→1, etc.
74 }
75};

81struct EliasDelta {
82 template<std::unsigned_integral T>
83 static void encode(T value, auto& sink) requires BitSink<decltype(sink)> {
84 ++value; // Map 0→1
85
86 int len = msb_position(value) + 1;
87
88 // Encode len using gamma
89 EliasGamma::encode(static_cast<T>(len - 1), sink);
90
91 // Write remaining bits of value (skip MSB which is always 1)
92 for (int i = len - 2; i >= 0; --i) {
93 sink.write((value >> i) & 1);
94 }
95 }

99 // Decode length
100 uint32_t len = EliasGamma::decode<uint32_t>(source) + 1;
101
102 // Read remaining bits
103 T value = 1;
104 for (uint32_t i = 0; i < len - 1; ++i) {
105 value = (value << 1) | source.read();
106 }
107
108 return value - 1;
109 }

112// ============================================================
113// Fibonacci Codec - Based on Fibonacci sequence
114// ============================================================
115
116struct Fibonacci {
117private:
118 static constexpr size_t MAX_FIBS = 92; // Enough for 64-bit
119
120 template<typename T>
121 static constexpr std::array<T, MAX_FIBS> compute_fibs() {
122 std::array<T, MAX_FIBS> fibs{};
123 fibs[0] = 1;
124 fibs[1] = 2;
125 for (size_t i = 2; i < MAX_FIBS && fibs[i-1] < std::numeric_limits<T>::max() / 2; ++i) {
126 fibs[i] = fibs[i-1] + fibs[i-2];
127 }
128 return fibs;
129 }
130
131public:
132 template<std::unsigned_integral T>
133 static void encode(T value, auto& sink) requires BitSink<decltype(sink)> {
134 if (value == 0) {
135 sink.write(true);
136 sink.write(true);
137 return;
138 }
139
140 constexpr auto fibs = compute_fibs<T>();
141 ++value; // Map to positive integers
142
143 // Find Fibonacci representation
144 std::array<bool, MAX_FIBS> bits{};
145 int highest = 0;
146
147 for (int i = MAX_FIBS - 1; i >= 0; --i) {
148 if (fibs[i] <= value && fibs[i] > 0) {
149 bits[i] = true;
150 value -= fibs[i];
151 highest = std::max(highest, i);
152 }
153 }
154
155 // Write bits from lowest to highest, then terminator
156 for (int i = 0; i <= highest; ++i) {
157 sink.write(bits[i]);
158 }
159 sink.write(true); // Terminator
160 }
161
162 template<std::unsigned_integral T>
163 static T decode(auto& source) requires BitSource<decltype(source)> {
164 constexpr auto fibs = compute_fibs<T>();
165
166 T value = 0;
167 bool prev_bit = false;
168
169 for (size_t i = 0; i < MAX_FIBS; ++i) {
170 bool bit = source.read();
171
172 if (bit && prev_bit) {
173 // Found terminator (two consecutive 1s)
174 return value > 0 ? value - 1 : 0;
175 }
176
177 if (bit) {
178 value += fibs[i];
179 }
180
181 prev_bit = bit;
182 }
183
184 return value > 0 ? value - 1 : 0;
185 }
186};

188// ============================================================
189// Rice Codec - Optimal for geometric distributions
190// ============================================================
191
192template<size_t K>
193struct Rice {
194 static_assert(K < 64, "K must be less than 64");
195
196 template<std::unsigned_integral T>
197 static void encode(T value, auto& sink) requires BitSink<decltype(sink)> {
198 T q = value >> K; // Quotient
199 T r = value & ((1u << K) - 1); // Remainder
200
201 // Encode quotient in unary
202 Unary::encode(q, sink);
203
204 // Encode remainder in binary
205 for (size_t i = 0; i < K; ++i) {
206 sink.write((r >> i) & 1);
207 }
208 }
209
210 template<std::unsigned_integral T>
211 static T decode(auto& source) requires BitSource<decltype(source)> {
212 // Decode quotient
213 T q = Unary::decode<T>(source);
214
215 // Decode remainder
216 T r = 0;
217 for (size_t i = 0; i < K; ++i) {
218 r |= static_cast<T>(source.read()) << i;
219 }
220
221 return (q << K) | r;
222 }
223};

252// ============================================================
253
254template<typename SmallCodec, typename LargeCodec, std::unsigned_integral T = uint32_t>
255struct Adaptive {
256 static constexpr T threshold = 128; // Tunable parameter
257
258 static void encode(T value, auto& sink) requires BitSink<decltype(sink)> {
259 if (value < threshold) {
260 sink.write(false); // Flag for small codec
261 SmallCodec::encode(value, sink);
262 } else {
263 sink.write(true); // Flag for large codec
264 LargeCodec::encode(value - threshold, sink);
265 }
266 }
267
268 static T decode(auto& source) requires BitSource<decltype(source)> {
269 if (!source.read()) {
270 return SmallCodec::template decode<T>(source);
271 } else {
272 return LargeCodec::template decode<T>(source) + threshold;
273 }
274 }
275};

292// Signed Integer Codec - Zigzag encoding
293// ============================================================
294
295template<typename UnsignedCodec>
296struct Signed {
297 template<std::signed_integral T>
298 static void encode(T value, auto& sink) requires BitSink<decltype(sink)> {
299 using U = std::make_unsigned_t<T>;
300 // Zigzag encoding: 0→0, -1→1, 1→2, -2→3, 2→4...
301 U encoded = (value < 0) ? ((-value) * 2 - 1) : (value * 2);
302 UnsignedCodec::encode(encoded, sink);
303 }
304
305 template<std::signed_integral T>
306 static T decode(auto& source) requires BitSource<decltype(source)> {
307 using U = std::make_unsigned_t<T>;
308 U encoded = UnsignedCodec::template decode<U>(source);
309 // Zigzag decoding
310 return (encoded & 1) ? -static_cast<T>((encoded + 1) / 2)
311 : static_cast<T>(encoded / 2);
312 }
313};

315// ============================================================
316// VByte (Variable Byte / Varint) - Industry standard codec
317// Protocol Buffers compatible, cache-friendly, SIMD-ready
318// ============================================================
319
320struct VByte {
321 template<std::unsigned_integral T, typename Sink>
322 static void encode(T value, Sink& sink) requires BitSink<Sink> {
323 // Each byte: 7 bits data (LSB first) + 1 continuation bit (MSB)
324 // Continuation bit: 0 = more bytes follow, 1 = last byte
325 do {
326 uint8_t byte = value & 0x7F;
327 value >>= 7;
328 if (value != 0) {
329 // More bytes to follow, continuation bit = 0
330 for (int i = 0; i < 8; ++i) {
331 sink.write((byte >> i) & 1);
332 }
333 } else {
334 // Last byte, continuation bit = 1
335 byte |= 0x80;
336 for (int i = 0; i < 8; ++i) {
337 sink.write((byte >> i) & 1);
338 }
339 }
340 } while (value != 0);
341 }
342
343 template<std::unsigned_integral T>
344 static T decode(auto& source) requires BitSource<decltype(source)> {
345 T result = 0;
346 int shift = 0;
347
348 while (true) {
349 // Read 8 bits
350 uint8_t byte = 0;
351 for (int i = 0; i < 8; ++i) {
352 byte |= static_cast<uint8_t>(source.read()) << i;
353 }
354
355 // Extract data bits (lower 7 bits)
356 T data = byte & 0x7F;
357 result |= data << shift;
358
359 // Check continuation bit (MSB)
360 if (byte & 0x80) {
361 // Last byte, we're done
362 break;
363 }
364
365 shift += 7;
366 }
367
368 return result;
369 }
370};

372// ============================================================
373// Exponential-Golomb - Parameterized family used in video coding
374// Order 0 = Elias Gamma, higher orders = flatter distribution
375// ============================================================
376
377template<size_t Order = 0>
378struct ExpGolomb {
379 static_assert(Order < 32, "Order must be less than 32");
380
381 template<std::unsigned_integral T>
382 static void encode(T value, auto& sink) requires BitSink<decltype(sink)> {
383 // Map input: value → value + 2^Order - 1
384 T mapped = value + (1u << Order) - 1;
385
386 // Compute q and r
387 T q = mapped >> Order; // quotient: mapped / 2^Order
388 T r = mapped & ((1u << Order) - 1); // remainder: mapped % 2^Order
389
390 // Encode q+1 in unary (length q, then a 1 bit)
391 int len = msb_position(q + 1) + 1;
392
393 // Write len-1 zeros
394 for (int i = 0; i < len - 1; ++i) {
395 sink.write(false);
396 }
397
398 // Write q+1 in binary (including implicit leading 1)
399 for (int i = len - 1; i >= 0; --i) {
400 sink.write(((q + 1) >> i) & 1);
401 }
402
403 // Write remainder in binary (Order bits)
404 for (size_t i = 0; i < Order; ++i) {
405 sink.write((r >> i) & 1);
406 }
407 }
408
409 template<std::unsigned_integral T>
410 static T decode(auto& source) requires BitSource<decltype(source)> {
411 // Count leading zeros
412 int zeros = 0;
413 while (!source.read()) {
414 ++zeros;
415 }
416
417 // Read zeros more bits to get q+1
418 T q_plus_1 = 1;
419 for (int i = 0; i < zeros; ++i) {
420 q_plus_1 = (q_plus_1 << 1) | source.read();
421 }
422 T q = q_plus_1 - 1;
423
424 // Read remainder (Order bits)
425 T r = 0;
426 for (size_t i = 0; i < Order; ++i) {
427 r |= static_cast<T>(source.read()) << i;
428 }
429
430 // Reconstruct mapped value
431 T mapped = (q << Order) | r;
432
433 // Unmap: mapped → value
434 return mapped - (1u << Order) + 1;
435 }
436};

443// ============================================================
444// Elias Omega - Recursive length encoding
445// More efficient than Delta for large integers
446// ============================================================
447
448struct EliasOmega {
449 template<std::unsigned_integral T>
450 static void encode(T value, auto& sink) requires BitSink<decltype(sink)> {
451 ++value; // Map 0→1, 1→2, etc.
452
453 // Build list of binary representations bottom-up
454 std::vector<std::pair<T, int>> stack;
455
456 while (value > 1) {
457 int len = msb_position(value) + 1;
458 stack.emplace_back(value, len);
459 value = len - 1; // Encode length-1 next
460 }
461
462 // Write components in reverse order (top-down)
463 for (auto it = stack.rbegin(); it != stack.rend(); ++it) {
464 auto [val, len] = *it;
465 // Write value in binary (MSB to LSB)
466 for (int i = len - 1; i >= 0; --i) {
467 sink.write((val >> i) & 1);
468 }
469 }
470
471 // Terminator: write a 0 bit
472 sink.write(false);
473 }
474
475 template<std::unsigned_integral T>
476 static T decode(auto& source) requires BitSource<decltype(source)> {
477 T n = 1; // Start with implicit 1
478
479 while (source.peek() && source.read()) {
480 // Read a 1 bit: we need to read more
481 // Current n tells us how many bits to read next
482 T next = 1; // Implicit leading 1
483
484 for (T i = 0; i < n; ++i) {
485 next = (next << 1) | source.read();
486 }
487
488 n = next;
489 }
490
491 // We've either read a 0 (terminator) or reached end
492 // n contains the decoded value + 1
493 return n - 1; // Map back: 1→0, 2→1, etc.
494 }
495};
