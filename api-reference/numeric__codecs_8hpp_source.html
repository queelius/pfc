17template<size_t MantissaBits = 23, size_t ExponentBits = 8, typename SignCodec = Boolean>
18struct FloatingPoint {
19 static_assert(MantissaBits > 0 && MantissaBits <= 52, "Invalid mantissa bits");
20 static_assert(ExponentBits > 0 && ExponentBits <= 11, "Invalid exponent bits");
21
22 // Encode a floating point number
23 static void encode(double value, auto& sink) requires BitSink<decltype(sink)> {
24 // Handle special cases
25 if (std::isnan(value)) {
26 sink.write(true); // Special flag
27 sink.write(true); // NaN marker
28 return;
29 }
30
31 if (std::isinf(value)) {
32 sink.write(true); // Special flag
33 sink.write(false); // Inf marker
34 SignCodec::encode(value < 0, sink);
35 return;
36 }
37
38 sink.write(false); // Not special
39
40 // Zero is special
41 if (value == 0.0) {
42 sink.write(true); // Zero flag
43 SignCodec::encode(std::signbit(value), sink); // Preserve -0.0
44 return;
45 }
46
47 sink.write(false); // Not zero
48
49 // Decompose the number
50 int exponent;
51 double mantissa = std::frexp(value, &exponent);
52
53 // Sign
54 bool negative = mantissa < 0;
55 SignCodec::encode(negative, sink);
56 if (negative) mantissa = -mantissa;
57
58 // Normalize mantissa to [0.5, 1)
59 // frexp already gives us this
60
61 // Encode exponent with bias
62 int32_t bias = (1 << (ExponentBits - 1)) - 1;
63 int32_t biased_exp = exponent + bias;
64
65 // Clamp to representable range
66 biased_exp = std::max(0, std::min((1 << ExponentBits) - 1, biased_exp));
67
68 for (size_t i = 0; i < ExponentBits; ++i) {
69 sink.write((biased_exp >> i) & 1);
70 }
71
72 // Encode mantissa (skip the implicit leading 1)
73 mantissa = (mantissa - 0.5) * 2; // Now in [0, 1)
74
75 for (size_t i = 0; i < MantissaBits; ++i) {
76 mantissa *= 2;
77 bool bit = mantissa >= 1.0;
78 sink.write(bit);
79 if (bit) mantissa -= 1.0;
80 }
81 }
82
83 static double decode(auto& source) requires BitSource<decltype(source)> {
84 // Check for special values
85 if (source.read()) { // Special flag
86 if (source.read()) { // NaN
87 return std::numeric_limits<double>::quiet_NaN();
88 } else { // Infinity
89 bool negative = SignCodec::decode(source);
90 return negative ? -std::numeric_limits<double>::infinity()
91 : std::numeric_limits<double>::infinity();
92 }
93 }
94
95 // Check for zero
96 if (source.read()) { // Zero flag
97 bool negative = SignCodec::decode(source);
98 return negative ? -0.0 : 0.0;
99 }
100
101 // Decode sign
102 bool negative = SignCodec::decode(source);
103
104 // Decode exponent
105 int32_t biased_exp = 0;
106 for (size_t i = 0; i < ExponentBits; ++i) {
107 biased_exp |= static_cast<int32_t>(source.read()) << i;
108 }
109
110 int32_t bias = (1 << (ExponentBits - 1)) - 1;
111 int exponent = biased_exp - bias;
112
113 // Decode mantissa
114 double mantissa = 0.5; // Implicit leading 1
115 double bit_value = 0.25;
116
117 for (size_t i = 0; i < MantissaBits; ++i) {
118 if (source.read()) {
119 mantissa += bit_value;
120 }
121 bit_value /= 2;
122 }
123
124 // Reconstruct the number
125 double result = std::ldexp(mantissa, exponent);
126 return negative ? -result : result;
127 }
128};

170template<typename NumeratorCodec = SignedGamma, typename DenominatorCodec = EliasGamma>
171struct Rational {
172 struct Fraction {
173 int64_t numerator;
174 uint64_t denominator;
175
176 // Simplify the fraction
177 void simplify() {
178 if (denominator == 0) {
179 denominator = 1; // Avoid division by zero
180 return;
181 }
182
183 uint64_t g = std::gcd(std::abs(numerator), denominator);
184 numerator /= g;
185 denominator /= g;
186 }
187
188 double to_double() const {
189 return static_cast<double>(numerator) / denominator;
190 }
191
192 static Fraction from_double(double value, uint64_t max_denominator = 1000000) {
193 // Use continued fractions to find best rational approximation
194 if (std::isnan(value) || std::isinf(value)) {
195 return {0, 1};
196 }
197
198 bool negative = value < 0;
199 value = std::abs(value);
200
201 uint64_t whole = static_cast<uint64_t>(value);
202 double frac = value - whole;
203
204 if (frac < 1e-10) {
205 return {negative ? -static_cast<int64_t>(whole) : whole, 1};
206 }
207
208 // Continued fraction expansion
209 uint64_t prev_num = 1, cur_num = whole;
210 uint64_t prev_den = 0, cur_den = 1;
211
212 while (cur_den < max_denominator && frac > 1e-10) {
213 double inv = 1.0 / frac;
214 uint64_t term = static_cast<uint64_t>(inv);
215 frac = inv - term;
216
217 uint64_t next_num = term * cur_num + prev_num;
218 uint64_t next_den = term * cur_den + prev_den;
219
220 if (next_den > max_denominator) break;
221
222 prev_num = cur_num;
223 cur_num = next_num;
224 prev_den = cur_den;
225 cur_den = next_den;
226 }
227
228 return {negative ? -static_cast<int64_t>(cur_num) : cur_num, cur_den};
229 }
230 };
231
232 static void encode(double value, auto& sink) requires BitSink<decltype(sink)> {
233 Fraction frac = Fraction::from_double(value);
234 frac.simplify();
235
236 NumeratorCodec::encode(frac.numerator, sink);
237 DenominatorCodec::encode(frac.denominator, sink);
238 }
239
240 static double decode(auto& source) requires BitSource<decltype(source)> {
241 int64_t num = NumeratorCodec::template decode<int64_t>(source);
242 uint64_t den = DenominatorCodec::template decode<uint64_t>(source);
243
244 if (den == 0) den = 1; // Safety
245 return static_cast<double>(num) / den;
246 }
247
248 // Alternative: encode/decode the Fraction directly
249 static void encode_fraction(const Fraction& frac, auto& sink) requires BitSink<decltype(sink)> {
250 Fraction simplified = frac;
251 simplified.simplify();
252
253 NumeratorCodec::encode(simplified.numerator, sink);
254 DenominatorCodec::encode(simplified.denominator, sink);
255 }
256
257 static Fraction decode_fraction(auto& source) requires BitSource<decltype(source)> {
258 int64_t num = NumeratorCodec::template decode<int64_t>(source);
259 uint64_t den = DenominatorCodec::template decode<uint64_t>(source);
260 return {num, den};
261 }
262};

268template<typename MantissaCodec = SignedGamma, typename ExponentCodec = SignedGamma>
269struct Scientific {
270 struct SciNumber {
271 double mantissa; // Typically in [-10, 10)
272 int32_t exponent;
273
274 double to_double() const {
275 return mantissa * std::pow(10.0, exponent);
276 }
277
278 static SciNumber from_double(double value) {
279 if (value == 0.0) return {0.0, 0};
280
281 int exp = 0;
282 double mant = value;
283
284 // Normalize to [-10, 10)
285 while (std::abs(mant) >= 10.0) {
286 mant /= 10.0;
287 exp++;
288 }
289 while (std::abs(mant) < 1.0 && mant != 0.0) {
290 mant *= 10.0;
291 exp--;
292 }
293
294 return {mant, exp};
295 }
296 };
297
298 static void encode(double value, auto& sink) requires BitSink<decltype(sink)> {
299 SciNumber sci = SciNumber::from_double(value);
300
301 // Encode mantissa as fixed decimal (3 digits precision)
302 int32_t mant_fixed = static_cast<int32_t>(std::round(sci.mantissa * 1000));
303 MantissaCodec::encode(mant_fixed, sink);
304
305 // Encode exponent
306 ExponentCodec::encode(sci.exponent, sink);
307 }
308
309 static double decode(auto& source) requires BitSource<decltype(source)> {
310 int32_t mant_fixed = MantissaCodec::template decode<int32_t>(source);
311 int32_t exp = ExponentCodec::template decode<int32_t>(source);
312
313 double mantissa = mant_fixed / 1000.0;
314 return mantissa * std::pow(10.0, exp);
315 }
