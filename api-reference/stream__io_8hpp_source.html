26// Custom streambuf for bit-oriented I/O
27class bit_streambuf : public std::streambuf {
28protected:
29 static constexpr std::size_t default_buffer_size = 8192;
30 std::vector<char> buffer_;
31 std::size_t bit_pos_ = 0;
32 uint8_t current_byte_ = 0;
33
34public:
35 explicit bit_streambuf(std::size_t buffer_size = default_buffer_size)
36 : buffer_(buffer_size) {
37 setp(buffer_.data(), buffer_.data() + buffer_.size() - 1);
38 }
39
40 // Flush any pending bits
41 void flush_bits() {
42 if (bit_pos_ > 0) {
43 *pptr() = static_cast<char>(current_byte_);
44 pbump(1);
45 bit_pos_ = 0;
46 current_byte_ = 0;
47 }
48 }
49
50 // Get/set bit position within current byte
51 std::size_t bit_position() const noexcept { return bit_pos_; }
52 void reset_bit_position() noexcept { bit_pos_ = 0; current_byte_ = 0; }
53};

55// Output bit streambuf - writes bits to an underlying ostream
56class output_bit_streambuf : public bit_streambuf {
57 std::ostream* stream_;
58
59protected:
60 int overflow(int ch) override {
61 if (stream_ && pptr() == epptr()) {
62 flush_bits();
63 stream_->write(pbase(), pptr() - pbase());
64 if (!stream_->good()) return traits_type::eof();
65 setp(buffer_.data(), buffer_.data() + buffer_.size() - 1);
66 }
67 if (ch != traits_type::eof()) {
68 *pptr() = static_cast<char>(ch);
69 pbump(1);
70 }
71 return ch;
72 }
73
74 int sync() override {
75 flush_bits();
76 if (stream_ && pptr() > pbase()) {
77 stream_->write(pbase(), pptr() - pbase());
78 if (!stream_->good()) return -1;
79 setp(buffer_.data(), buffer_.data() + buffer_.size() - 1);
80 }
81 if (stream_) stream_->flush();
82 return 0;
83 }
84
85public:
86 explicit output_bit_streambuf(std::ostream& os, std::size_t buffer_size = default_buffer_size)
87 : bit_streambuf(buffer_size), stream_(&os) {}
88
89 ~output_bit_streambuf() {
90 sync();
91 }
92};

94// Input bit streambuf - reads bits from an underlying istream
95class input_bit_streambuf : public bit_streambuf {
96 std::istream* stream_;
97
98protected:
99 int underflow() override {
100 if (stream_ && gptr() == egptr()) {
101 stream_->read(buffer_.data(), buffer_.size());
102 auto bytes_read = stream_->gcount();
103 if (bytes_read == 0) return traits_type::eof();
104 setg(buffer_.data(), buffer_.data(), buffer_.data() + bytes_read);
105 }
106 return gptr() < egptr() ? traits_type::to_int_type(*gptr()) : traits_type::eof();
107 }
108
109public:
110 explicit input_bit_streambuf(std::istream& is, std::size_t buffer_size = default_buffer_size)
111 : bit_streambuf(buffer_size), stream_(&is) {
112 setg(buffer_.data(), buffer_.data(), buffer_.data());
113 }
114};

117// Stream Bit Writer - BitSink that writes to std::ostream
118// ============================================================
119
120class stream_bit_writer {
121 std::unique_ptr<output_bit_streambuf> buf_;
122 std::ostream* stream_;
123 uint8_t byte_ = 0;
124 uint8_t bit_pos_ = 0;
125 std::size_t bytes_written_ = 0;
126
127public:
128 explicit stream_bit_writer(std::ostream& os, std::size_t buffer_size = 8192)
129 : buf_(std::make_unique<output_bit_streambuf>(os, buffer_size))
130 , stream_(&os) {}
131
132 stream_bit_writer(stream_bit_writer&&) = default;
133 stream_bit_writer& operator=(stream_bit_writer&&) = default;
134
135 // BitSink interface
136 void write(bool bit) {
137 byte_ |= (bit ? 1u : 0u) << bit_pos_;
138 if (++bit_pos_ == 8) {
139 stream_->put(static_cast<char>(byte_));
140 byte_ = 0;
141 bit_pos_ = 0;
142 ++bytes_written_;
143 }
144 }
145
146 // Write multiple bits at once (LSB first)
147 void write_bits(uint64_t bits, std::size_t count) {
148 for (std::size_t i = 0; i < count; ++i) {
149 write((bits >> i) & 1);
150 }
151 }
152
153 // Write bytes directly (bypassing bit-level operations)
154 void write_bytes(const uint8_t* data, std::size_t size) {
155 align(); // Ensure byte alignment first
156 stream_->write(reinterpret_cast<const char*>(data), size);
157 bytes_written_ += size;
158 }
159
160 // Align to byte boundary
161 void align() {
162 if (bit_pos_ > 0) {
163 stream_->put(static_cast<char>(byte_));
164 byte_ = 0;
165 bit_pos_ = 0;
166 ++bytes_written_;
167 }
168 }
169
170 // Flush the underlying stream
171 void flush() {
172 align();
173 stream_->flush();
174 }
175
176 // Get statistics
177 [[nodiscard]] std::size_t bytes_written() const noexcept {
178 return bytes_written_ + (bit_pos_ > 0 ? 1 : 0);
179 }
180
181 [[nodiscard]] bool good() const noexcept { return stream_->good(); }
182 [[nodiscard]] bool fail() const noexcept { return stream_->fail(); }
183 [[nodiscard]] bool bad() const noexcept { return stream_->bad(); }
184 [[nodiscard]] bool eof() const noexcept { return stream_->eof(); }
185 [[nodiscard]] explicit operator bool() const noexcept { return good(); }
186
187 ~stream_bit_writer() {
188 if (stream_ && stream_->good()) {
189 flush();
190 }
191 }
192};

194// ============================================================
195// Stream Bit Reader - BitSource that reads from std::istream
196// ============================================================
197
198class stream_bit_reader {
199 std::unique_ptr<input_bit_streambuf> buf_;
200 std::istream* stream_;
201 uint8_t byte_ = 0;
202 uint8_t bit_pos_ = 8; // Start at 8 to force initial read
203 std::size_t bytes_read_ = 0;
204 bool eof_reached_ = false;
205
206public:
207 explicit stream_bit_reader(std::istream& is, std::size_t buffer_size = 8192)
208 : buf_(std::make_unique<input_bit_streambuf>(is, buffer_size))
209 , stream_(&is) {}
210
211 stream_bit_reader(stream_bit_reader&&) = default;
212 stream_bit_reader& operator=(stream_bit_reader&&) = default;
213
214 // BitSource interface
215 [[nodiscard]] bool peek() const noexcept {
216 return !eof_reached_ && (bit_pos_ < 8 || stream_->good());
217 }
218
219 bool read() {
220 if (bit_pos_ == 8) {
221 if (!stream_->good()) {
222 eof_reached_ = true;
223 return false;
224 }
225 int ch = stream_->get();
226 if (ch == std::char_traits<char>::eof()) {
227 eof_reached_ = true;
228 return false;
229 }
230 byte_ = static_cast<uint8_t>(ch);
231 bit_pos_ = 0;
232 ++bytes_read_;
233 }
234 return (byte_ >> bit_pos_++) & 1;
235 }
236
237 // Read multiple bits at once (LSB first)
238 [[nodiscard]] uint64_t read_bits(std::size_t count) {
239 uint64_t result = 0;
240 for (std::size_t i = 0; i < count; ++i) {
241 result |= static_cast<uint64_t>(read()) << i;
242 }
243 return result;
244 }
245
246 // Read bytes directly (bypassing bit-level operations)
247 std::size_t read_bytes(uint8_t* data, std::size_t size) {
248 align(); // Ensure byte alignment first
249 stream_->read(reinterpret_cast<char*>(data), size);
250 auto bytes = stream_->gcount();
251 bytes_read_ += bytes;
252 return bytes;
253 }
254
255 // Skip to next byte boundary
256 void align() {
257 bit_pos_ = 8;
258 }
259
260 // Get statistics
261 [[nodiscard]] std::size_t bytes_read() const noexcept {
262 return bytes_read_;
263 }
264
265 [[nodiscard]] bool good() const noexcept { return stream_->good() && !eof_reached_; }
266 [[nodiscard]] bool fail() const noexcept { return stream_->fail(); }
267 [[nodiscard]] bool bad() const noexcept { return stream_->bad(); }
268 [[nodiscard]] bool eof() const noexcept { return eof_reached_ || stream_->eof(); }
269 [[nodiscard]] explicit operator bool() const noexcept { return good(); }
270};

301// Write compressed data to file
302template<typename T, typename Codec = codecs::EliasGamma, typename Allocator = std::allocator<T>>
303result<std::size_t> write_compressed_file(const std::string& filename,
304 const std::vector<T, Allocator>& data,
305 std::ios::openmode mode = std::ios::binary) {
306 std::ofstream file(filename, mode | std::ios::binary);
307 if (!file) {
308 return make_error_code(pfc_error::io_error);
309 }
310
311 try {
312 stream_bit_writer writer(file);
313
314 // Write header with size
315 Codec::encode(static_cast<uint64_t>(data.size()), writer);
316
317 // Write data
318 for (const auto& item : data) {
319 Codec::encode(item, writer);
320 }
321
322 writer.flush();
323 std::size_t bytes = writer.bytes_written();
324
325 if (!writer.good()) {
326 return make_error_code(pfc_error::io_error);
327 }
328
329 return bytes;
330 } catch (const std::exception&) {
331 return make_error_code(pfc_error::compression_error);

335// Read compressed data from file
336template<typename T, typename Codec = codecs::EliasGamma, typename Allocator = std::allocator<T>>
337result<std::vector<T, Allocator>> read_compressed_file(const std::string& filename,
338 std::ios::openmode mode = std::ios::binary) {

344 try {

350 // Read data

352 data.reserve(size);

355 data.push_back(Codec::template decode<T>(reader));
356 }

359 return make_error_code(pfc_error::corrupted_data);

361
362 return data;
363 } catch (const std::exception&) {
364 return make_error_code(pfc_error::decompression_error);
