67// Compress data to bytes
68template<typename T, typename Codec = EliasGamma>
69std::vector<uint8_t> compress(const std::vector<T>& data) {
70 if (data.empty()) return {};
71
72 // Estimate size (conservative)
73 size_t estimated_bytes = data.size() * sizeof(T) / 2 + 128;
74 std::vector<uint8_t> buffer(estimated_bytes);
75
76 BitWriter writer{std::span<uint8_t>{buffer}};
77
78 // Write count
79 EliasGamma::encode(static_cast<uint32_t>(data.size()), writer);
80
81 // Write values
82 for (const auto& value : data) {
83 Codec::encode(value, writer);
84 }
85
86 writer.align();
87
88 // Resize to actual size
89 buffer.resize(writer.bytes_written());
90 return buffer;
91}

93// Decompress bytes to data
94template<typename T, typename Codec = EliasGamma>
95std::vector<T> decompress(const std::vector<uint8_t>& compressed) {
96 if (compressed.empty()) return {};
97
98 BitReader reader(compressed);
99
100 // Read count
101 uint32_t count = EliasGamma::template decode<uint32_t>(reader);
102
103 // Read values
104 std::vector<T> result;
105 result.reserve(count);
106
107 for (uint32_t i = 0; i < count; ++i) {
108 result.push_back(Codec::template decode<T>(reader));
109 }
110
111 return result;
112}

120 Geometric, // P(n) ∝ p^n -> Rice or Elias Gamma

148// Calculate entropy of data
149template<typename T>
150double calculate_entropy(const std::vector<T>& data) {
151 if (data.empty()) return 0.0;
152
153 std::unordered_map<T, size_t> freq;
154 for (const auto& val : data) {
155 ++freq[val];
156 }
157
158 double entropy = 0.0;
159 double n = static_cast<double>(data.size());
160
161 for (const auto& [_, count] : freq) {
162 double p = count / n;
163 if (p > 0) {
164 entropy -= p * std::log2(p);
165 }
166 }
167
168 return entropy;
169}

171// Estimate best codec based on data sample
172template<typename T>
173const char* suggest_codec(const std::vector<T>& sample) {
174 if (sample.empty()) return "Unknown";
175
176 // Calculate statistics
177 auto [min_it, max_it] = std::minmax_element(sample.begin(), sample.end());
178 T min_val = *min_it;
179 T max_val = *max_it;
180 T range = max_val - min_val;
181
182 // Count unique values
183 std::unordered_set<T> unique(sample.begin(), sample.end());
184 double unique_ratio = static_cast<double>(unique.size()) / sample.size();
185
186 // Calculate average value
187 double avg = std::accumulate(sample.begin(), sample.end(), 0.0) / sample.size();
188
189 // Simple heuristics
190 if (unique_ratio < 0.1) {
191 return "RunLength";
192 } else if (range < 256) {
193 return "Fixed<8>";
194 } else if (avg < 100 && max_val < 1000) {
195 return "EliasGamma";
196 } else if (max_val > 1'000'000) {
197 return "EliasDelta";
198 } else {
199 return "EliasGamma"; // Good default
200 }
201}
