174// ============================================================
175// Safe BitWriter with bounds checking
176// ============================================================
177
178template<typename Allocator = std::allocator<uint8_t>>
179class SafeBitWriter {
180public:
181 using allocator_type = Allocator;
182 using byte_allocator = typename std::allocator_traits<Allocator>::template rebind_alloc<uint8_t>;
183 using buffer_type = std::vector<uint8_t, byte_allocator>;
184
185 explicit SafeBitWriter(size_t max_size, const Allocator& alloc = Allocator())
186 : buffer_(alloc), byte_(0), bit_(0), max_size_(max_size) {
187 buffer_.reserve(std::min(size_t(64), max_size));
188 }
189
190 result<void> write_bit(bool b) {
191 if (buffer_.size() >= max_size_ && bit_ == 0) {
192 return make_error_code(pfc_error::buffer_overflow);
193 }
194
195 if (b) byte_ |= 1u << bit_;
196 if (++bit_ == 8) {
197 buffer_.push_back(byte_);
198 byte_ = 0;
199 bit_ = 0;
200 }
201 return {};
202 }
203
204 result<void> write_bits(uint64_t v, int n) {
205 for (int i = 0; i < n; ++i) {
206 if (auto res = write_bit((v >> i) & 1); !res) {
207 return res;
208 }
209 }
210 return {};
211 }
212
213 result<void> align() {
214 if (bit_ > 0) {
215 if (buffer_.size() >= max_size_) {
216 return make_error_code(pfc_error::buffer_overflow);
217 }
218 buffer_.push_back(byte_);
219 byte_ = 0;
220 bit_ = 0;
221 }
222 return {};
223 }
224
225 const buffer_type& buffer() const { return buffer_; }
226 size_t size() const { return buffer_.size() + (bit_ > 0 ? 1 : 0); }
227 size_t bit_count() const { return buffer_.size() * 8 + bit_; }
228 size_t max_size() const { return max_size_; }
229
230 allocator_type get_allocator() const {
231 return buffer_.get_allocator();
232 }
233
234private:
235 buffer_type buffer_;
236 uint8_t byte_;
237 int bit_;
238 size_t max_size_;
239};

242// Safe BitReader with validation

245template<typename Allocator = std::allocator<uint8_t>>
246class SafeBitReader {
247public:
248 using allocator_type = Allocator;
249
250 SafeBitReader(const uint8_t* data, size_t size)
251 : data_(data), size_(size), pos_(0), byte_(0), bit_(8) {
252 if (!data_ && size_ > 0) {
253 valid_ = false;
254 }
255 }
256
257 SafeBitReader(const std::vector<uint8_t, Allocator>& buffer)
258 : data_(buffer.data()), size_(buffer.size()), pos_(0), byte_(0), bit_(8) {}
259
260 result<bool> read_bit() {
261 if (!valid_) {
262 return make_error_code(pfc_error::invalid_input);
263 }
264
265 if (bit_ == 8) {
266 if (pos_ >= size_) {
267 return make_error_code(pfc_error::eof_reached);
268 }
269 byte_ = data_[pos_++];
270 bit_ = 0;
271 }
272 return (byte_ >> bit_++) & 1u;
273 }
274
275 result<uint64_t> read_bits(int n) {
276 if (n > 64) {
277 return make_error_code(pfc_error::invalid_parameter);
278 }
279
280 uint64_t v = 0;
281 for (int i = 0; i < n; ++i) {
282 auto bit_res = read_bit();
283 if (!bit_res) {
284 return bit_res.error();
285 }
286 v |= uint64_t(*bit_res) << i;
287 }
288 return v;
289 }
290
291 void align() {
292 bit_ = 8;
293 }
294
295 bool eof() const {
296 return pos_ >= size_ && bit_ == 8;
297 }
298
299 size_t position() const {
300 return pos_ * 8 + bit_;
301 }
302
303 bool is_valid() const { return valid_; }
304
305private:
306 const uint8_t* data_;
307 size_t size_;
308 size_t pos_;
309 uint8_t byte_;
310 int bit_;
311 bool valid_ = true;
312};
