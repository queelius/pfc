83private:

87 template<typename Sink>
88 struct EncodeVisitor {
89 Sink& sink;
90
91 template<typename T>
92 void operator()(const T& value) const {
93 T::encode(value, sink);
94 }

96
97 // Helper to get index bits needed
98 static constexpr size_t index_bits() {
99 if (type_count <= 2) return 1;
100 if (type_count <= 4) return 2;
101 if (type_count <= 8) return 3;
102 if (type_count <= 16) return 4;
103 if (type_count <= 32) return 5;
104 if (type_count <= 64) return 6;
105 return 7; // Up to 128 variants
106 }
107
108public:
109 // Constructors
110 PackedVariant() = default;

112 template<typename T>

114
115 // Type-safe construction from value_type
116 explicit PackedVariant(const value_type& val) {
117 std::visit([this](const auto& v) {
118 using V = std::decay_t<decltype(v)>;
119 // Find matching packed type
120 set_variant<0, V>(v);
121 }, val);
122 }
123
124private:
125 template<size_t I, typename V>
126 void set_variant(const V& v) {
127 if constexpr (I < sizeof...(Types)) {
128 using T = std::tuple_element_t<I, std::tuple<Types...>>;
129 if constexpr (std::is_same_v<typename T::value_type, V>) {
130 data_.template emplace<T>(v);
131 } else {
132 set_variant<I + 1, V>(v);
133 }
134 }
135 }

137public:

139 template<typename Visitor>

141 return std::visit(std::forward<Visitor>(vis), data_);
142 }

147 }

152 }

169 });

172 // Serialization
173 template<typename Sink>
174 static void encode(const PackedVariant& var, Sink& sink) requires BitSink<Sink> {
175 // Encode type index using minimal bits
176 size_t idx = var.index();
177 for (size_t i = 0; i < index_bits(); ++i) {
178 sink.write((idx >> i) & 1);
179 }
180
181 // Encode the value
182 std::visit(EncodeVisitor<Sink>{sink}, var.data_);
183 }
184
185 template<typename Source>
186 static PackedVariant decode(Source& source) requires BitSource<Source> {
187 // Decode type index
188 size_t idx = 0;
189 for (size_t i = 0; i < index_bits(); ++i) {
190 idx |= static_cast<size_t>(source.read()) << i;
191 }
192
193 // Decode the appropriate type
194 return decode_index<0>(idx, source);
195 }
196
197private:
198 template<size_t I, typename Source>
199 static PackedVariant decode_index(size_t idx, Source& source) {
200 if constexpr (I < sizeof...(Types)) {
201 if (I == idx) {
202 using T = std::tuple_element_t<I, std::tuple<Types...>>;
203 return PackedVariant{T::decode(source)};
204 } else {
205 return decode_index<I + 1>(idx, source);
206 }
207 } else {
208 throw std::runtime_error("Invalid variant index");
209 }
210 }

269class PackedList {
270public:
271 using value_type = std::vector<typename Element::value_type>;
272 using element_type = Element;
273
274private:
275 PackedListNode<Element> node_;
276
277public:
278 // Constructors
279 PackedList() : node_(Unit{}) {} // Empty list
280
281 PackedList(const Element& head, std::shared_ptr<PackedList> tail)
282 : node_(PackedPair<Element, PackedSharedPtr<PackedList>>{
283 head, PackedSharedPtr<PackedList>{tail}}) {}
284
285 // Build from vector
286 explicit PackedList(const value_type& vec) {
287 if (vec.empty()) {
288 node_ = PackedListNode<Element>{Unit{}};
289 } else {
290 // Build list from back to front
291 std::shared_ptr<PackedList> tail = std::make_shared<PackedList>();
292 for (auto it = vec.rbegin(); it != vec.rend(); ++it) {
293 auto new_list = std::make_shared<PackedList>();
294 new_list->node_ = PackedListNode<Element>{
295 PackedPair<Element, PackedSharedPtr<PackedList>>{
296 Element{*it},
297 PackedSharedPtr<PackedList>{tail}
298 }
299 };
300 tail = new_list;
301 }
302 *this = *tail;
303 }
304 }
305
306 // Query operations
307 [[nodiscard]] bool empty() const noexcept {
308 return node_.index() == 0; // Unit is at index 0
309 }
310
311 [[nodiscard]] const Element* head() const {
312 if (auto* cons = node_.template get_if<PackedPair<Element, PackedSharedPtr<PackedList>>>()) {
313 return &cons->first();
314 }
315 return nullptr;
316 }
317
318 [[nodiscard]] std::shared_ptr<PackedList> tail() const {
319 if (auto* cons = node_.template get_if<PackedPair<Element, PackedSharedPtr<PackedList>>>()) {
320 return cons->second().value();
321 }
322 return nullptr;
323 }
324
325 // Convert to vector
326 [[nodiscard]] value_type value() const {
327 value_type result;
328 const PackedList* current = this;
329
330 while (!current->empty()) {
331 if (auto* h = current->head()) {
332 result.push_back(h->value());
333 }
334 if (auto t = current->tail()) {
335 current = t.get();
336 } else {
337 break;
338 }
339 }
340
341 return result;
342 }
343
344 // Functional operations
345 template<typename F>
346 auto map(F&& f) const {
347 using NewElement = Packed<decltype(f(std::declval<typename Element::value_type>())),
348 typename Element::codec_type>;
349
350 if (empty()) {
351 return PackedList<NewElement>{};
352 }
353
354 NewElement new_head{f(head()->value())};
355 auto new_tail = tail()->map(f);
356 return PackedList<NewElement>{new_head, std::make_shared<PackedList<NewElement>>(new_tail)};
357 }

360 void for_each(F&& f) const {
361 if (!empty()) {
362 f(*head());

364 t->for_each(f);
365 }
366 }
367 }
