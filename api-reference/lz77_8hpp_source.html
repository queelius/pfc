31// LZ77 Match Structure
32// ============================================================
33
34struct LZ77Match {
35 uint16_t distance; // Distance back in window (0 = literal)
36 uint16_t length; // Match length (0 for literals)
37 uint8_t literal; // Literal byte (when distance == 0)
38
39 bool is_literal() const { return distance == 0; }
40
41 // Encode the match to bits
42 template<typename BitSink>
43 result<void> encode(BitSink& sink) const {
44 if (is_literal()) {
45 // Encode as literal: 0 bit + 8 bits for byte
46 sink.write(false);
47 for (int i = 0; i < 8; ++i) {
48 sink.write((literal >> i) & 1);
49 }
50 } else {
51 // Encode as match: 1 bit + distance + length
52 sink.write(true);
53 // Encode distance (16 bits)
54 for (int i = 0; i < 16; ++i) {
55 sink.write((distance >> i) & 1);
56 }
57 // Encode length (8 bits)
58 for (int i = 0; i < 8; ++i) {
59 sink.write((length >> i) & 1);
60 }
61 }
62 return {};
63 }
64
65 // Decode match from bits
66 template<typename BitSource>
67 static result<LZ77Match> decode(BitSource& source) {
68 if (!source.peek()) {
69 return make_error_code(pfc_error::eof_reached);
70 }
71
72 LZ77Match match{};
73 bool is_match = source.read();
74
75 if (!is_match) {
76 // Decode literal
77 match.distance = 0;
78 match.literal = 0;
79 for (int i = 0; i < 8; ++i) {
80 if (!source.peek()) {
81 return make_error_code(pfc_error::incomplete_data);
82 }
83 match.literal |= source.read() << i;
84 }
85 } else {
86 // Decode match
87 match.distance = 0;
88 for (int i = 0; i < 16; ++i) {
89 if (!source.peek()) {
90 return make_error_code(pfc_error::incomplete_data);
91 }
92 match.distance |= source.read() << i;
93 }
94
95 match.length = 0;
96 for (int i = 0; i < 8; ++i) {
97 if (!source.peek()) {
98 return make_error_code(pfc_error::incomplete_data);
99 }
100 match.length |= source.read() << i;
101 }
102 }
103
104 return match;
105 }
106};

112template<typename Allocator = std::allocator<uint8_t>>
113class LZ77Compressor {
114public:
115 using allocator_type = Allocator;
116
117 explicit LZ77Compressor(const LZ77Config& config = LZ77Config::default_config(),
118 const Allocator& alloc = Allocator())
119 : config_(config), window_(alloc), lookahead_(alloc) {
120 // Note: deque doesn't have reserve(), but that's okay
121 // It will grow as needed
122 }
123
124 // Compress data
125 template<typename InputIt, typename OutputIt>
126 result<size_t> compress(InputIt first, InputIt last, OutputIt output) {
127 size_t match_count = 0;
128
129 // Fill lookahead buffer initially
130 for (size_t i = 0; i < config_.lookahead_size && first != last; ++i, ++first) {
131 lookahead_.push_back(*first);
132 }
133
134 while (!lookahead_.empty()) {
135 auto match = find_best_match();
136
137 if (match.is_literal() || match.length < config_.min_match_length) {
138 // Output literal
139 *output++ = LZ77Match{0, 0, lookahead_[0]};
140 advance_window(1);
141 } else {
142 // Output match
143 *output++ = match;
144 advance_window(match.length);
145 }
146
147 match_count++;
148
149 // Refill lookahead
150 while (lookahead_.size() < config_.lookahead_size && first != last) {
151 lookahead_.push_back(*first++);
152 }
153 }
154
155 return match_count;
156 }
157
158 // Compress to bit stream
159 template<typename InputIt, typename BitSink>
160 result<size_t> compress_to_bits(InputIt first, InputIt last, BitSink& sink) {
161 std::vector<LZ77Match> matches;
162 auto res = compress(first, last, std::back_inserter(matches));
163 if (!res) return res.error();
164
165 // Write header: number of matches
166 uint32_t num_matches = matches.size();
167 for (int i = 0; i < 32; ++i) {
168 sink.write((num_matches >> i) & 1);
169 }
170
171 // Write matches
172 for (const auto& match : matches) {
173 if (auto enc_res = match.encode(sink); !enc_res) {
174 return enc_res.error();
175 }
176 }
177
178 return matches.size();
179 }
180
181private:
182 LZ77Config config_;
183 std::deque<uint8_t, Allocator> window_;
184 std::deque<uint8_t, Allocator> lookahead_;
185
186 LZ77Match find_best_match() {
187 if (lookahead_.empty()) {
188 return {0, 0, 0};
189 }
190
191 LZ77Match best_match{0, 0, lookahead_[0]};
192
193 // Search for matches in the window
194 for (size_t i = 0; i < window_.size(); ++i) {
195 size_t match_length = 0;
196
197 // Check how long the match is
198 while (match_length < lookahead_.size() &&
199 match_length < config_.max_match_length &&
200 i + match_length < window_.size() &&
201 window_[i + match_length] == lookahead_[match_length]) {
202 match_length++;
203 }
204
205 // Update best match if this is better
206 if (match_length >= config_.min_match_length && match_length > best_match.length) {
207 best_match.distance = window_.size() - i;
208 best_match.length = match_length;
209 }
210 }
211
212 return best_match;
213 }
214
215 void advance_window(size_t count) {
216 for (size_t i = 0; i < count && !lookahead_.empty(); ++i) {
217 if (window_.size() >= config_.window_size) {
218 window_.pop_front();
219 }
220 window_.push_back(lookahead_.front());
221 lookahead_.pop_front();
222 }
223 }
224};

230template<typename Allocator = std::allocator<uint8_t>>
231class LZ77Decompressor {
232public:
233 using allocator_type = Allocator;
234
235 explicit LZ77Decompressor(const Allocator& alloc = Allocator())
236 : output_(alloc) {
237 output_.reserve(65536); // Reserve initial space
238 }
239
240 // Decompress matches
241 template<typename InputIt, typename OutputIt>
242 result<size_t> decompress(InputIt first, InputIt last, OutputIt output) {
243 output_.clear();
244 size_t output_size = 0;
245
246 for (auto it = first; it != last; ++it) {
247 const LZ77Match& match = *it;
248
249 if (match.is_literal()) {
250 // Output literal
251 *output++ = match.literal;
252 output_.push_back(match.literal);
253 output_size++;
254 } else {
255 // Output match from history
256 if (match.distance > output_.size()) {
257 return make_error_code(pfc_error::corrupted_data);
258 }
259
260 size_t start = output_.size() - match.distance;
261 for (size_t i = 0; i < match.length; ++i) {
262 uint8_t byte = output_[start + i];
263 *output++ = byte;
264 output_.push_back(byte);
265 output_size++;
266 }
267 }
268 }
269
270 return output_size;
271 }
272
273 // Decompress from bit stream
274 template<typename BitSource, typename OutputIt>
275 result<size_t> decompress_from_bits(BitSource& source, OutputIt output) {
276 // Read header: number of matches
277 uint32_t num_matches = 0;
278 for (int i = 0; i < 32; ++i) {
279 if (!source.peek()) {
280 return make_error_code(pfc_error::incomplete_data);
281 }
282 num_matches |= source.read() << i;
283 }
284
285 if (num_matches > 10000000) { // Sanity check
286 return make_error_code(pfc_error::corrupted_data);
287 }
288
289 std::vector<LZ77Match> matches;
290 matches.reserve(num_matches);
291
292 // Read matches
293 for (uint32_t i = 0; i < num_matches; ++i) {
294 auto match_res = LZ77Match::decode(source);
295 if (!match_res) {
296 return match_res.error();
297 }
298 matches.push_back(*match_res);
299 }
300
301 return decompress(matches.begin(), matches.end(), output);
302 }
303
304private:
305 std::vector<uint8_t, Allocator> output_;
306};

308// ============================================================
309// LZSS (Lempel-Ziv-Storer-Szymanski) - LZ77 variant
310// ============================================================
311
312class LZSS {
313public:
314 // LZSS uses a more efficient encoding than plain LZ77
315 // It uses a flag byte to indicate which of the next 8 items are matches vs literals
316
317 template<typename InputIt, typename BitSink>
318 static result<size_t> compress(InputIt first, InputIt last, BitSink& sink) {
319 LZ77Compressor compressor(LZ77Config::balanced());
320 std::vector<LZ77Match> matches;
321
322 auto res = compressor.compress(first, last, std::back_inserter(matches));
323 if (!res) return res.error();
324
325 // Group matches into blocks of 8 with flag bytes
326 size_t i = 0;
327 while (i < matches.size()) {
328 uint8_t flags = 0;
329 size_t block_size = std::min(size_t(8), matches.size() - i);
330
331 // Set flags for this block
332 for (size_t j = 0; j < block_size; ++j) {
333 if (!matches[i + j].is_literal()) {
334 flags |= (1 << j);
335 }
336 }
337
338 // Write flag byte
339 for (int b = 0; b < 8; ++b) {
340 sink.write((flags >> b) & 1);
341 }
342
343 // Write block items
344 for (size_t j = 0; j < block_size; ++j) {
345 const auto& match = matches[i + j];
346 if (match.is_literal()) {
347 // Write literal byte
348 for (int b = 0; b < 8; ++b) {
349 sink.write((match.literal >> b) & 1);
350 }
351 } else {
352 // Write distance (12 bits) and length (4 bits) - more compact
353 for (int b = 0; b < 12; ++b) {
354 sink.write((match.distance >> b) & 1);
355 }
356 uint8_t encoded_length = std::min<uint8_t>(match.length - 3, 15);
357 for (int b = 0; b < 4; ++b) {
358 sink.write((encoded_length >> b) & 1);
359 }
360 }
361 }
362
363 i += block_size;
364 }
365
366 return matches.size();
367 }
368};
