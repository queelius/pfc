19template<typename T>
20concept Regular = std::copyable<T> && std::equality_comparable<T>;
21
22template<typename T>
23concept TotallyOrdered = Regular<T> && std::totally_ordered<T>;
24
25template<typename T>
26concept Additive = Regular<T> && requires(T a, T b) {
27 { a + b } -> std::convertible_to<T>;
28 { a - b } -> std::convertible_to<T>;
29 { -a } -> std::convertible_to<T>;
30 { a += b } -> std::same_as<T&>;
31 { a -= b } -> std::same_as<T&>;
32};
33
34template<typename T, typename Scalar>
35concept VectorSpace = Additive<T> && requires(T v, Scalar s) {
36 { v * s } -> std::convertible_to<T>;
37 { s * v } -> std::convertible_to<T>;
38 { v / s } -> std::convertible_to<T>;
39 { v *= s } -> std::same_as<T&>;
40 { v /= s } -> std::same_as<T&>;
41};
