18template<typename P>
19concept PackedValue = requires(const P& p, BitWriter& w, BitReader& r) {
20 typename P::value_type;
21 { P::encode(p, w) } -> std::same_as<void>;
22 { P::decode(r) } -> std::same_as<P>;
23 { p.value() } -> std::convertible_to<typename P::value_type>;
24};

397 // Write a packed value
398 template<PackedValue P>
399 PackedBuffer& operator<<(const P& packed) {
400 BitWriter writer(data_.data() + data_.size());
401 P::encode(packed, writer);
402
403 // Update size
404 size_t bytes_written = writer.position() - (data_.data() + data_.size());
405 size_t new_bits = bytes_written * 8;
406
407 if (writer.bytes_written() > 0) {
408 data_.push_back(0); // Add partial byte
409 new_bits = (bytes_written - 1) * 8 + (8 - __builtin_clz(data_.back()));
410 }
411
412 data_.resize(data_.size() + bytes_written);
413 bit_size_ += new_bits;
414
415 return *this;
416 }

420 P extract() const {
421 BitReader reader(data_);
422 return P::decode(reader);
423 }
