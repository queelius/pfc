20template<typename Element, typename Container>
21class PackedProxy {
22public:
23 using value_type = typename Element::value_type;
24
25private:
26 Container* container_;
27 size_t index_;
28 mutable std::optional<Element> cached_;
29
30 void ensure_cached() const {
31 if (!cached_) {
32 // Decode from container at index
33 BitReader reader(container_->raw_data(index_));
34 cached_ = Element::decode(reader);
35 }
36 }
37
38public:
39 PackedProxy(Container* c, size_t i) : container_(c), index_(i) {}
40
41 // Conversion to value type
42 operator value_type() const {
43 ensure_cached();
44 return cached_->value();
45 }
46
47 // Assignment from value type
48 PackedProxy& operator=(const value_type& val) {
49 cached_ = Element{val};
50
51 // Re-encode to container
52 container_->update_element(index_, *cached_);
53 return *this;
54 }
55
56 // Access to packed element
57 const Element& get_packed() const {
58 ensure_cached();
59 return *cached_;
60 }
61
62 // Comparison operators
63 bool operator==(const PackedProxy& other) const {
64 return value_type(*this) == value_type(other);
65 }
66
67 bool operator<(const PackedProxy& other) const {
68 return value_type(*this) < value_type(other);
69 }
70};

179template<PackedValue Element>
180class PackedContainer {
181public:
182 using value_type = typename Element::value_type;
183 using size_type = size_t;
184 using difference_type = std::ptrdiff_t;
185 using iterator = PackedIterator<Element, PackedContainer>;
186 using const_iterator = PackedIterator<Element, const PackedContainer>;
187
188private:
189 // Store offsets for O(1) random access
190 std::vector<uint8_t> data_;
191 std::vector<size_t> offsets_;
192 size_t count_ = 0;
193
194public:
195 // Constructors
196 PackedContainer() = default;
197
198 template<typename InputIt>
199 PackedContainer(InputIt first, InputIt last) {
200 for (auto it = first; it != last; ++it) {
201 push_back(Element{*it});
202 }
203 }
204
205 // Size operations
206 [[nodiscard]] size_type size() const noexcept { return count_; }
207 [[nodiscard]] bool empty() const noexcept { return count_ == 0; }
208
209 void reserve(size_type new_cap) {
210 offsets_.reserve(new_cap + 1);
211 // Estimate data size (heuristic)
212 data_.reserve(new_cap * 8);
213 }
214
215 void clear() noexcept {
216 data_.clear();
217 offsets_.clear();
218 count_ = 0;
219 }
220
221 // Element access
222 [[nodiscard]] value_type operator[](size_type pos) const {
223 BitReader reader(data_.data() + offsets_[pos],
224 offsets_[pos + 1] - offsets_[pos]);
225 return Element::decode(reader).value();
226 }
227
228 [[nodiscard]] value_type at(size_type pos) const {
229 if (pos >= count_) {
230 throw std::out_of_range("PackedContainer::at");
231 }
232 return (*this)[pos];
233 }
234
235 [[nodiscard]] value_type front() const { return (*this)[0]; }
236 [[nodiscard]] value_type back() const { return (*this)[count_ - 1]; }
237
238 // Modifiers
239 void push_back(const Element& elem) {
240 // Add start offset only for the first element
241 if (offsets_.empty()) {
242 offsets_.push_back(0);
243 }
244
245 // Encode to temporary buffer
246 std::vector<uint8_t> temp(64); // Should be enough
247 BitWriter writer(temp.data());
248 Element::encode(elem, writer);
249 writer.align();
250
251 // Append to data
252 size_t bytes = writer.bytes_written();
253 data_.insert(data_.end(), temp.begin(), temp.begin() + bytes);
254
255 ++count_;
256
257 // Always push end offset (which becomes start of next element)
258 offsets_.push_back(data_.size());
259 }
260
261 void push_back(Element&& elem) {
262 push_back(elem); // Same as const version for now
263 }
264
265 template<typename... Args>
266 void emplace_back(Args&&... args) {
267 push_back(Element{std::forward<Args>(args)...});
268 }
269
270 void pop_back() {
271 if (count_ > 0) {
272 --count_;
273 data_.resize(offsets_[count_]);
274 offsets_.resize(count_ + 1);
275 }
276 }
277
278 // Iterators
279 iterator begin() noexcept { return iterator(this, 0); }
280 iterator end() noexcept { return iterator(this, count_); }
281 const_iterator begin() const noexcept { return const_iterator(this, 0); }
282 const_iterator end() const noexcept { return const_iterator(this, count_); }
283 const_iterator cbegin() const noexcept { return begin(); }
284 const_iterator cend() const noexcept { return end(); }
285
286 // Internal access for proxy
287 std::span<const uint8_t> raw_data(size_t index) const {
288 return std::span<const uint8_t>(data_.data() + offsets_[index],
289 offsets_[index + 1] - offsets_[index]);
290 }
291
292 void update_element(size_t index, const Element& elem) {
293 // For simplicity, rebuild from this point
294 // In production, use a more sophisticated approach
295
296 // Save elements after index
297 std::vector<Element> saved;
298 for (size_t i = index + 1; i < count_; ++i) {
299 BitReader reader(data_.data() + offsets_[i],
300 offsets_[i + 1] - offsets_[i]);
301 saved.push_back(Element::decode(reader));
302 }
303
304 // Truncate to before index
305 count_ = index;
306 data_.resize(offsets_[index]);
307 offsets_.resize(index + 1);
308
309 // Re-add updated element and saved elements
310 push_back(elem);
311 for (const auto& e : saved) {
312 push_back(e);
313 }
314 }
315
316 // Capacity
317 [[nodiscard]] size_type capacity() const noexcept {
318 return offsets_.capacity();
319 }
320
321 [[nodiscard]] size_type data_bytes() const noexcept {
322 return data_.size();
323 }
324
325 // Compression ratio
326 [[nodiscard]] double compression_ratio() const noexcept {
327 if (count_ == 0) return 1.0;
328 size_t unpacked_size = count_ * sizeof(value_type);
329 return static_cast<double>(unpacked_size) / data_.size();
330 }

441// Transform that produces a new packed container
442template<PackedValue Input, typename F>
443auto packed_transform(const PackedContainer<Input>& input, F&& f) {
444 using OutputValue = decltype(f(std::declval<typename Input::value_type>()));
445 using Output = Packed<OutputValue, typename Input::codec_type>;
446
447 PackedContainer<Output> result;
448 result.reserve(input.size());
449
450 for (size_t i = 0; i < input.size(); ++i) {
451 result.push_back(Output{f(input[i])});
452 }
453
454 return result;
455}

480// Parallel transform using execution policies
481template<typename ExecPolicy, PackedValue Input, typename F>
482auto packed_transform_par(ExecPolicy&& policy,
483 const PackedContainer<Input>& input,
484 F&& f) {
485 using OutputValue = decltype(f(std::declval<typename Input::value_type>()));
486 using Output = Packed<OutputValue, typename Input::codec_type>;
487
488 PackedContainer<Output> result;
489 result.reserve(input.size());
490
491 // Decode in parallel, encode sequentially (for now)
492 std::vector<OutputValue> temp(input.size());
493
494 std::transform(policy,
495 input.begin(), input.end(),
496 temp.begin(),
497 [&f](const auto& proxy) { return f(typename Input::value_type(proxy)); });
498
499 for (const auto& val : temp) {
500 result.push_back(Output{val});
501 }
502
503 return result;
504}

506// Zero-copy merge of sorted packed containers
507template<PackedValue Element>
508PackedContainer<Element> packed_merge(const PackedContainer<Element>& a,
509 const PackedContainer<Element>& b) {
510 PackedContainer<Element> result;
511 result.reserve(a.size() + b.size());
512
513 size_t i = 0, j = 0;
514 while (i < a.size() && j < b.size()) {
515 if (a[i] <= b[j]) {
516 // Direct copy of packed bytes (zero-copy)
517 auto data = a.raw_data(i);
518 BitReader reader(data.data(), data.size());
519 result.push_back(Element::decode(reader));
520 ++i;
521 } else {
522 auto data = b.raw_data(j);
523 BitReader reader(data.data(), data.size());
524 result.push_back(Element::decode(reader));
525 ++j;
526 }
527 }
528
529 // Copy remaining elements
530 while (i < a.size()) {
531 auto data = a.raw_data(i);
532 BitReader reader(data.data(), data.size());
533 result.push_back(Element::decode(reader));
534 ++i;
535 }
536
537 while (j < b.size()) {
538 auto data = b.raw_data(j);
539 BitReader reader(data.data(), data.size());
540 result.push_back(Element::decode(reader));
541 ++j;
542 }
543
544 return result;
545}

556 // Extract values, sort, rebuild

566 container.clear();
