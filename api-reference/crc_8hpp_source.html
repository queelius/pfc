309template<typename Checksum = CRC32>
310class ChecksummedData {
311public:
312 template<typename InputIt, typename OutputIt>
313 static void add_checksum(InputIt first, InputIt last, OutputIt output) {
314 // Copy data and calculate checksum
315 Checksum checksum;
316 for (auto it = first; it != last; ++it) {
317 uint8_t byte = *it;
318 checksum.update(byte);
319 *output++ = byte;
320 }
321
322 // Append checksum
323 auto sum = checksum.finalize();
324 for (size_t i = 0; i < sizeof(sum); ++i) {
325 *output++ = static_cast<uint8_t>((sum >> (i * 8)) & 0xFF);
326 }
327 }
328
329 template<typename InputIt>
330 static bool verify_checksum(InputIt first, InputIt last) {
331 Checksum temp_checksum;
332 auto checksum_size = sizeof(decltype(temp_checksum.finalize()));
333
334 if (std::distance(first, last) < checksum_size) {
335 return false;
336 }
337
338 // Calculate checksum of data
339 Checksum checksum;
340 auto data_end = last - checksum_size;
341 for (auto it = first; it != data_end; ++it) {
342 checksum.update(*it);
343 }
344
345 // Read stored checksum
346 decltype(checksum.finalize()) stored_sum = 0;
347 size_t shift = 0;
348 for (auto it = data_end; it != last; ++it) {
349 stored_sum |= static_cast<decltype(stored_sum)>(*it) << shift;
350 shift += 8;
351 }
352
353 return checksum.finalize() == stored_sum;
354 }
355};
