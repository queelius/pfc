18template<typename T>
19concept BitSink = requires(T& sink, bool bit) {
20 { sink.write(bit) } -> std::same_as<void>;
21};
22
23template<typename T>
24concept BitSource = requires(T& source) {
25 { source.read() } -> std::same_as<bool>;
26 { source.peek() } -> std::convertible_to<bool>; // Check if more bits available
27};

29template<typename C, typename T, typename Sink>
30concept Encoder = BitSink<Sink> && requires(const T& value, Sink& sink) {
31 { C::encode(value, sink) } -> std::same_as<void>;
32};

34template<typename C, typename T, typename Source>
35concept Decoder = BitSource<Source> && requires(Source& source) {

54class BitWriter {
55 uint8_t* ptr_;
56 uint8_t* start_; // Track start position
57 uint8_t byte_ = 0;
58 uint8_t bit_pos_ = 0;
59
60public:
61 explicit BitWriter(uint8_t* ptr) noexcept : ptr_(ptr), start_(ptr) {}
62 explicit BitWriter(std::span<uint8_t> buffer) noexcept : ptr_(buffer.data()), start_(buffer.data()) {}
63
64 void write(bool bit) noexcept {
65 byte_ |= (bit ? 1u : 0u) << bit_pos_;
66 if (++bit_pos_ == 8) {
67 *ptr_++ = byte_;
68 byte_ = 0;
69 bit_pos_ = 0;
70 }
71 }

73 // Write multiple bits at once (LSB first)
74 void write_bits(uint64_t bits, size_t count) noexcept {
75 for (size_t i = 0; i < count; ++i) {
76 write((bits >> i) & 1);
77 }
78 }

86 }
87 }

105class BitReader {
106 const uint8_t* ptr_;
107 const uint8_t* end_;
108 uint8_t byte_ = 0;
109 uint8_t bit_pos_ = 8; // Start at 8 to force initial read
110
111public:
112 BitReader(const uint8_t* ptr, size_t size) noexcept
113 : ptr_(ptr), end_(ptr + size) {}
114
115 explicit BitReader(std::span<const uint8_t> buffer) noexcept
116 : ptr_(buffer.data()), end_(buffer.data() + buffer.size()) {}
117
118 [[nodiscard]] bool peek() const noexcept {
119 return ptr_ < end_ || bit_pos_ < 8;
120 }
121
122 bool read() noexcept {
123 if (bit_pos_ == 8) {
124 if (ptr_ >= end_) return false; // EOF
125 byte_ = *ptr_++;
126 bit_pos_ = 0;
127 }
128 return (byte_ >> bit_pos_++) & 1;
129 }
130
131 // Read multiple bits at once (LSB first)
132 [[nodiscard]] uint64_t read_bits(size_t count) noexcept {
133 uint64_t result = 0;
134 for (size_t i = 0; i < count; ++i) {
135 result |= static_cast<uint64_t>(read()) << i;
136 }
137 return result;
138 }
139
140 // Skip to next byte boundary
141 void align() noexcept {
142 bit_pos_ = 8;
143 }
144
145 // Get current position
146 [[nodiscard]] const uint8_t* position() const noexcept {
147 return ptr_ - (bit_pos_ < 8 ? 1 : 0);
148 }
149};

155// Count leading zeros (portable, efficient)
156template<std::unsigned_integral T>
157[[nodiscard]] constexpr int count_leading_zeros(T x) noexcept {
158 if (x == 0) return std::numeric_limits<T>::digits;
159 if constexpr (sizeof(T) <= sizeof(unsigned int)) {
160 return __builtin_clz(x) - (32 - std::numeric_limits<T>::digits);
161 } else if constexpr (sizeof(T) <= sizeof(unsigned long)) {
162 return __builtin_clzl(x) - (64 - std::numeric_limits<T>::digits);
163 } else {
164 return __builtin_clzll(x) - (64 - std::numeric_limits<T>::digits);
165 }
166}
167
168// Most significant bit position (0-based)
169template<std::unsigned_integral T>
170[[nodiscard]] constexpr int msb_position(T x) noexcept {
171 return x == 0 ? -1 : std::numeric_limits<T>::digits - 1 - count_leading_zeros(x);
172}
