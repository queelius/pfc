152class BasicPackedBuffer {
153public:
154 using allocator_type = Allocator;
155 using buffer_type = std::vector<uint8_t, Allocator>;
156
157 explicit BasicPackedBuffer(const Allocator& alloc = Allocator())
158 : data_(alloc) {}
159
160 template<typename T, typename Codec>
161 void pack(const T& value) {
162 BasicBitWriter<Allocator> writer(get_allocator());
163 Codec::encode(value, writer);
164 writer.align();
165
166 // Append to our buffer
167 const auto& buf = writer.buffer();
168 data_.insert(data_.end(), buf.begin(), buf.end());
169 }
170
171 template<typename T, typename Codec>
172 T unpack(size_t& offset) const {
173 BasicBitReader<Allocator> reader(data_.data() + offset, data_.size() - offset);
174 T result = Codec::template decode<T>(reader);
175 offset += (reader.position() + 7) / 8; // Round up to byte boundary
176 return result;
177 }
178
179 const buffer_type& data() const { return data_; }
180 size_t size() const { return data_.size(); }
181
182 allocator_type get_allocator() const {
183 return data_.get_allocator();
184 }
185
186 void clear() { data_.clear(); }
187 void reserve(size_t n) { data_.reserve(n); }

189private:
190 buffer_type data_;
191};
