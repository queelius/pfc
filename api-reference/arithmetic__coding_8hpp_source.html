94// Adaptive probability model that learns from data
95template<std::size_t Symbols>
96class adaptive_probability_model {
97 std::array<uint32_t, Symbols> frequencies_;
98 std::array<uint32_t, Symbols + 1> cumulative_;
99 uint32_t total_;
100 static constexpr uint32_t max_frequency = 16383; // Prevent overflow
101 static constexpr uint32_t rescale_threshold = max_frequency * Symbols;
102
103public:
104 adaptive_probability_model() : frequencies_{}, cumulative_{}, total_(0) {
105 reset();
106 }
107
108 void reset() {
109 std::fill(frequencies_.begin(), frequencies_.end(), 1);
110 build_cumulative();
111 }
112
113 [[nodiscard]] std::size_t symbol_count() const noexcept { return Symbols; }
114
115 [[nodiscard]] uint32_t frequency(std::size_t symbol) const noexcept {
116 return frequencies_[symbol];
117 }
118
119 [[nodiscard]] uint32_t cumulative_frequency(std::size_t symbol) const noexcept {
120 return cumulative_[symbol];
121 }
122
123 [[nodiscard]] uint32_t total_frequency() const noexcept {
124 return total_;
125 }
126
127 [[nodiscard]] std::size_t find_symbol(uint32_t value) const noexcept {
128 // Linear search for small alphabets, binary search for large ones
129 if constexpr (Symbols <= 16) {
130 for (std::size_t i = 0; i < Symbols; ++i) {
131 if (value < cumulative_[i + 1]) return i;
132 }
133 return Symbols - 1;
134 } else {
135 auto it = std::upper_bound(cumulative_.begin(), cumulative_.end() - 1, value);
136 return std::distance(cumulative_.begin(), it) - 1;
137 }
138 }
139
140 void update(std::size_t symbol) {
141 frequencies_[symbol]++;
142
143 // Update cumulative frequencies from symbol onward
144 for (std::size_t i = symbol + 1; i <= Symbols; ++i) {
145 cumulative_[i]++;
146 }
147 total_++;
148
149 // Rescale if necessary to prevent overflow
150 if (total_ >= rescale_threshold) {
151 rescale();
152 }
153 }
154
155private:
156 void build_cumulative() {
157 cumulative_[0] = 0;
158 for (std::size_t i = 0; i < Symbols; ++i) {
159 cumulative_[i + 1] = cumulative_[i] + frequencies_[i];
160 }
161 total_ = cumulative_[Symbols];
162 }
163
164 void rescale() {
165 // Halve all frequencies (keeping minimum of 1)
166 for (auto& freq : frequencies_) {
167 freq = (freq + 1) / 2;
168 }
169 build_cumulative();
170 }
171};

173// Context model for order-N modeling
174template<std::size_t Symbols, std::size_t Order>
175class context_model {
176 static constexpr std::size_t context_count = 1u << (Order * 8);
177 std::vector<adaptive_probability_model<Symbols>> models_;
178 std::size_t current_context_ = 0;
179
180public:
181 context_model() : models_(context_count) {}
182
183 void reset() {
184 for (auto& model : models_) {
185 model.reset();
186 }
187 current_context_ = 0;
188 }
189
190 void set_context(const uint8_t* context_bytes) {
191 current_context_ = 0;
192 for (std::size_t i = 0; i < Order; ++i) {
193 current_context_ = (current_context_ << 8) | context_bytes[i];
194 }
195 current_context_ %= context_count;
196 }

199 [[nodiscard]] const auto& current_model() const { return models_[current_context_]; }

202// ============================================================
203// Arithmetic Encoder - The heart of optimal compression
204// ============================================================
205
206class arithmetic_encoder {
207 static constexpr uint32_t top_value = 0x7FFFFFFF; // 31 bits
208 static constexpr uint32_t first_quarter = top_value / 4 + 1;
209 static constexpr uint32_t half = 2 * first_quarter;
210 static constexpr uint32_t third_quarter = 3 * first_quarter;
211
212 BitWriter* sink_;
213 uint32_t low_ = 0;
214 uint32_t high_ = top_value;
215 uint32_t bits_to_follow_ = 0;
216
217 void output_bit_plus_follow(bool bit) {
218 sink_->write(bit);
219 while (bits_to_follow_ > 0) {
220 sink_->write(!bit);
221 bits_to_follow_--;
222 }
223 }
224
225public:
226 explicit arithmetic_encoder(BitWriter& sink) : sink_(&sink) {}
227
228 template<ProbabilityModel Model>
229 void encode_symbol(std::size_t symbol, Model& model) {
230 uint32_t range = high_ - low_ + 1;
231 uint32_t total = model.total_frequency();
232
233 // Ensure we have a valid total frequency
234 if (total == 0) {
235 return; // Can't encode with empty model
236 }
237
238 uint32_t cum_freq = model.cumulative_frequency(symbol);
239 uint32_t freq = model.frequency(symbol);
240
241 // Update range based on symbol probability
242 // Use uint64_t to prevent overflow in intermediate calculations
243 uint64_t r = range;
244 uint32_t new_high = low_ + static_cast<uint32_t>((r * (cum_freq + freq)) / total) - 1;
245 uint32_t new_low = low_ + static_cast<uint32_t>((r * cum_freq) / total);
246
247 // Ensure high_ > low_ (prevent range collapse)
248 if (new_high <= new_low) {
249 new_high = new_low + 1;
250 }
251
252 high_ = new_high;
253 low_ = new_low;
254
255 // Normalize and output bits
256 while (true) {
257 if (high_ < half) {
258 output_bit_plus_follow(false);
259 low_ *= 2;
260 high_ = high_ * 2 + 1;
261 } else if (low_ >= half) {
262 output_bit_plus_follow(true);
263 low_ = (low_ - half) * 2;
264 high_ = (high_ - half) * 2 + 1;
265 } else if (low_ >= first_quarter && high_ < third_quarter) {
266 bits_to_follow_++;
267 low_ = (low_ - first_quarter) * 2;
268 high_ = (high_ - first_quarter) * 2 + 1;
269 } else {
270 break;
271 }
272 }
273
274 model.update(symbol);
275 }
276
277 void finish() {
278 // Output final bits
279 bits_to_follow_++;
280 if (low_ < first_quarter) {
281 output_bit_plus_follow(false);
282 } else {
283 output_bit_plus_follow(true);
284 }
285 }

292class arithmetic_decoder {
293 static constexpr uint32_t top_value = 0x7FFFFFFF;
294 static constexpr uint32_t first_quarter = top_value / 4 + 1;
295 static constexpr uint32_t half = 2 * first_quarter;
296 static constexpr uint32_t third_quarter = 3 * first_quarter;
297
298 BitReader* source_;
299 uint32_t low_ = 0;
300 uint32_t high_ = top_value;
301 uint32_t value_ = 0;
302
303 bool read_bit() {
304 return source_->read();
305 }
306
307public:
308 explicit arithmetic_decoder(BitReader& source) : source_(&source) {
309 // Initialize decoder with first 31 bits
310 for (int i = 0; i < 31; ++i) {
311 value_ = (value_ << 1) | (read_bit() ? 1 : 0);
312 }
313 }
314
315 template<ProbabilityModel Model>
316 std::size_t decode_symbol(Model& model) {
317 uint32_t range = high_ - low_ + 1;
318 uint32_t total = model.total_frequency();
319
320 // Ensure we have a valid total frequency
321 if (total == 0) {
322 return 0; // Can't decode with empty model
323 }
324
325 // Ensure range is non-zero to prevent division by zero
326 if (range == 0) {
327 return 0; // Invalid state
328 }
329
330 // Use uint64_t to prevent overflow in intermediate calculations
331 // The encoder computes: new_low = low_ + (range * cum_freq) / total
332 // To invert: cum_freq = ((value_ - low_) * total) / range
333 uint64_t v = value_ - low_;
334 uint32_t scaled_value = static_cast<uint32_t>((v * total) / range);
335
336 // Find symbol
337 std::size_t symbol = model.find_symbol(scaled_value);
338
339 // Update range
340 uint32_t cum_freq = model.cumulative_frequency(symbol);
341 uint32_t freq = model.frequency(symbol);
342
343 uint64_t r = range;
344 uint32_t new_high = low_ + static_cast<uint32_t>((r * (cum_freq + freq)) / total) - 1;
345 uint32_t new_low = low_ + static_cast<uint32_t>((r * cum_freq) / total);
346
347 // Ensure high_ > low_ (prevent range collapse)
348 if (new_high <= new_low) {
349 new_high = new_low + 1;
350 }
351
352 high_ = new_high;
353 low_ = new_low;
354
355 // Normalize and read new bits
356 while (true) {
357 if (high_ < half) {
358 low_ *= 2;
359 high_ = high_ * 2 + 1;
360 value_ = (value_ * 2) | (read_bit() ? 1 : 0);
361 } else if (low_ >= half) {
362 low_ = (low_ - half) * 2;
363 high_ = (high_ - half) * 2 + 1;
364 value_ = ((value_ - half) * 2) | (read_bit() ? 1 : 0);
365 } else if (low_ >= first_quarter && high_ < third_quarter) {
366 low_ = (low_ - first_quarter) * 2;
367 high_ = (high_ - first_quarter) * 2 + 1;
368 value_ = ((value_ - first_quarter) * 2) | (read_bit() ? 1 : 0);
369 } else {
370 break;
371 }
372 }
373
374 model.update(symbol);
375 return symbol;
376 }

383class range_encoder {
384 static constexpr uint64_t top = 1ULL << 56; // Use 56-bit range
385 static constexpr uint64_t bottom = 1ULL << 48; // Normalize when range < 2^48
386
387 BitWriter* sink_;
388 uint64_t low_ = 0;
389 uint64_t range_ = top;
390
391 void normalize() {
392 while (range_ < bottom) {
393 // Output top byte
394 sink_->write_bits((low_ >> 56) & 0xFF, 8);
395 low_ <<= 8;
396 range_ <<= 8;
397 }
398 }
399
400public:
401 explicit range_encoder(BitWriter& sink) : sink_(&sink) {}
402
403 template<ProbabilityModel Model>
404 void encode_symbol(std::size_t symbol, Model& model) {
405 uint32_t total = model.total_frequency();
406
407 if (total == 0) {
408 return; // Can't encode with empty model
409 }
410
411 uint32_t cum_freq = model.cumulative_frequency(symbol);
412 uint32_t freq = model.frequency(symbol);
413
414 // Simple 64-bit arithmetic, no overflow possible
415 low_ += (range_ * cum_freq) / total;
416 range_ = (range_ * freq) / total;
417
418 if (range_ == 0) range_ = 1;
419
420 normalize();
421 model.update(symbol);
422 }
423
424 void encode_bit(bool bit, uint32_t prob_zero) {
425 uint32_t prob_scale = 0x10000;
426
427 if (!bit) {
428 range_ = (range_ * prob_zero) / prob_scale;
429 } else {
430 low_ += (range_ * prob_zero) / prob_scale;
431 range_ = (range_ * (prob_scale - prob_zero)) / prob_scale;
432 }
433
434 if (range_ == 0) range_ = 1;
435 normalize();
436 }
437
438 void finish() {
439 // Output remaining bytes (7 bytes for 56-bit value)
440 for (int i = 0; i < 7; ++i) {
441 sink_->write_bits((low_ >> 56) & 0xFF, 8);
442 low_ <<= 8;
443 }
444 }

447class range_decoder {
448 static constexpr uint64_t top = 1ULL << 56; // Match encoder
449 static constexpr uint64_t bottom = 1ULL << 48; // Match encoder
450
451 BitReader* source_;
452 uint64_t low_ = 0;
453 uint64_t code_ = 0;
454 uint64_t range_ = top;
455
456 void normalize() {
457 while (range_ < bottom) {
458 code_ = (code_ << 8) | source_->read_bits(8);
459 low_ <<= 8;
460 range_ <<= 8;
461 }
462 }
463
464public:
465 explicit range_decoder(BitReader& source) : source_(&source) {
466 // Initialize with first 7 bytes (56 bits)
467 for (int i = 0; i < 7; ++i) {
468 code_ = (code_ << 8) | source_->read_bits(8);

472 template<ProbabilityModel Model>
473 std::size_t decode_symbol(Model& model) {
474 uint32_t total = model.total_frequency();
475
476 if (total == 0) {
477 return 0; // Can't decode with empty model
478 }
479
480 // Find symbol by computing scaled value
481 // Use the same formula as encoder to avoid rounding errors
482 uint64_t helper = ((code_ - low_) + 1) * total - 1;
483 uint64_t scaled_value = helper / range_;
484
485 // Find symbol
486 std::size_t symbol = model.find_symbol(static_cast<uint32_t>(scaled_value));
487
488 // Update range - must match encoder exactly
489 uint32_t cum_freq = model.cumulative_frequency(symbol);
490 uint32_t freq = model.frequency(symbol);
491
492 low_ += (range_ * cum_freq) / total;
493 range_ = (range_ * freq) / total;
494
495 if (range_ == 0) range_ = 1; // Prevent range collapse
496
497 normalize();
498 model.update(symbol);
499 return symbol;
500 }
501
502 bool decode_bit(uint32_t prob_zero) {
503 uint32_t prob_scale = 0x10000;
504
505 uint64_t threshold = low_ + (range_ * prob_zero) / prob_scale;
506
507 bool bit = (code_ >= threshold);
508
509 if (!bit) {
510 range_ = (range_ * prob_zero) / prob_scale;
511 } else {
512 low_ = threshold;
513 range_ = (range_ * (prob_scale - prob_zero)) / prob_scale;
514 }
515
516 if (range_ == 0) range_ = 1;
517 normalize();
518 return bit;
519 }
