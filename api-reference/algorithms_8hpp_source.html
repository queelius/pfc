77struct CompressionStats {
78 size_t original_bits;
79 size_t compressed_bits;
80 double ratio;
81 double bits_per_value;
82};
83
84template<typename T, typename Codec>
85CompressionStats analyze_compression(const std::vector<T>& values) {
86 // Calculate original size (assuming fixed width)
87 size_t original_bits = values.size() * sizeof(T) * 8;
88
89 // Calculate compressed size
90 std::vector<uint8_t> buffer;
91 buffer.reserve(original_bits / 8 + 1024); // Reasonable initial size
92
93 BitWriter writer(buffer.data());
94 for (const auto& value : values) {
95 Codec::encode(value, writer);
96 }
97 writer.align();
98
99 size_t compressed_bits = (writer.position() - buffer.data()) * 8;
100
101 return CompressionStats{
102 .original_bits = original_bits,
103 .compressed_bits = compressed_bits,
104 .ratio = static_cast<double>(original_bits) / compressed_bits,
105 .bits_per_value = static_cast<double>(compressed_bits) / values.size()
106 };
107}

123 std::tuple<CodecInfo<Codecs>...> infos{
124 CodecInfo<Codecs>{analyze_compression<T, Codecs>(sample)}...
125 };
126
127 // Find codec with best compression ratio
128 auto best_codec = std::apply([](const auto&... info) {
129 const void* best = nullptr;
130 double best_ratio = 0.0;
131
132 ((info.stats.ratio > best_ratio ?
133 (best_ratio = info.stats.ratio, best = &info) : 0), ...);

135 return best;
136 }, infos);

138 return best_codec;

146template<PackedValue P>
147class PackedStream {
148private:
149 BitReader reader_;
150
151public:
152 explicit PackedStream(std::span<const uint8_t> data)
153 : reader_(data) {}
154
155 // Read next value
156 std::optional<P> next() {
157 if (!reader_.peek()) {
158 return std::nullopt;
159 }
160 return P::decode(reader_);
161 }
162
163 // Read n values
164 std::vector<P> take(size_t n) {
165 std::vector<P> result;
166 result.reserve(n);
167
168 for (size_t i = 0; i < n && reader_.peek(); ++i) {
169 result.push_back(P::decode(reader_));
170 }
171
172 return result;
173 }
174
175 // Skip n values
176 void skip(size_t n) {
177 for (size_t i = 0; i < n && reader_.peek(); ++i) {
178 P::decode(reader_);
179 }
180 }
181
182 // Check if more values available
183 [[nodiscard]] bool has_more() const noexcept {
184 return reader_.peek();
185 }
186};

192template<typename T, typename Codec>
193class DeltaCodec {
194public:
195 template<typename Sink>
196 static void encode_deltas(const std::vector<T>& values, Sink& sink) requires BitSink<Sink> {
197 if (values.empty()) return;
198
199 // Encode first value directly
200 Codec::encode(values[0], sink);
201
202 // Encode deltas
203 for (size_t i = 1; i < values.size(); ++i) {
204 T delta = values[i] - values[i-1];
205 if constexpr (std::is_signed_v<T>) {
206 codecs::SignedGamma::encode(delta, sink);
207 } else {
208 Codec::encode(delta, sink);
209 }
210 }
211 }
212
213 template<typename Source>
214 static std::vector<T> decode_deltas(Source& source, size_t count) requires BitSource<Source> {
215 if (count == 0) return {};
216
217 std::vector<T> result;
218 result.reserve(count);
219
220 // Decode first value
221 result.push_back(Codec::template decode<T>(source));
222
223 // Decode deltas and reconstruct values
224 for (size_t i = 1; i < count; ++i) {
225 T delta;
226 if constexpr (std::is_signed_v<T>) {
227 delta = codecs::SignedGamma::template decode<T>(source);
228 } else {
229 delta = Codec::template decode<T>(source);
230 }
231 result.push_back(result.back() + delta);
232 }
233
234 return result;
235 }
236};

238// ============================================================
239// Run-Length Encoding
240// ============================================================
241
242template<typename T, typename ValueCodec, typename CountCodec = codecs::EliasGamma>
243class RunLengthCodec {
244public:
245 template<typename Sink>
246 static void encode(const std::vector<T>& values, Sink& sink) requires BitSink<Sink> {
247 if (values.empty()) {
248 CountCodec::encode(0u, sink);
249 return;
250 }
251
252 // Count runs
253 std::vector<std::pair<T, uint32_t>> runs;
254 T current = values[0];
255 uint32_t count = 1;
256
257 for (size_t i = 1; i < values.size(); ++i) {
258 if (values[i] == current) {
259 ++count;
260 } else {
261 runs.emplace_back(current, count);
262 current = values[i];
263 count = 1;
264 }
265 }
266 runs.emplace_back(current, count);
267
268 // Encode number of runs
269 CountCodec::encode(static_cast<uint32_t>(runs.size()), sink);
270
271 // Encode runs
272 for (const auto& [value, run_count] : runs) {
273 ValueCodec::encode(value, sink);
274 CountCodec::encode(run_count - 1, sink); // -1 since count >= 1
275 }
276 }
277
278 template<typename Source>
279 static std::vector<T> decode(Source& source) requires BitSource<Source> {
280 uint32_t num_runs = CountCodec::template decode<uint32_t>(source);
281
282 std::vector<T> result;
283 for (uint32_t i = 0; i < num_runs; ++i) {
284 T value = ValueCodec::template decode<T>(source);
285 uint32_t count = CountCodec::template decode<uint32_t>(source) + 1;
286
287 for (uint32_t j = 0; j < count; ++j) {
288 result.push_back(value);
289 }
290 }
291
292 return result;
293 }
294};

297// Parallel Encoding (for large datasets)
298// ============================================================
299
300template<typename T, typename Codec>
301struct ParallelEncoder {
302 static std::vector<uint8_t> encode(const std::vector<T>& values,
303 size_t chunk_size = 1024) {
304 if (values.empty()) return {};
305
306 // Split into chunks
307 std::vector<std::vector<uint8_t>> chunks;
308 chunks.reserve((values.size() + chunk_size - 1) / chunk_size);
309
310 for (size_t i = 0; i < values.size(); i += chunk_size) {
311 size_t end = std::min(i + chunk_size, values.size());
312
313 // Encode chunk
314 std::vector<uint8_t> chunk_buffer;
315 chunk_buffer.resize(chunk_size * sizeof(T)); // Worst case
316
317 BitWriter writer(chunk_buffer.data());
318 for (size_t j = i; j < end; ++j) {
319 Codec::encode(values[j], writer);
320 }
321 writer.align();
322
323 chunk_buffer.resize(writer.position() - chunk_buffer.data());
324 chunks.push_back(std::move(chunk_buffer));
325 }
326
327 // Combine chunks
328 std::vector<uint8_t> result;
329 size_t total_size = 0;
330 for (const auto& chunk : chunks) {
331 total_size += chunk.size();
332 }
333
334 result.reserve(total_size + chunks.size() * 4); // Space for sizes
335
336 // Write chunk count
337 BitWriter header_writer(result.data());
338 codecs::EliasGamma::encode(static_cast<uint32_t>(chunks.size()),
339 header_writer);
340
341 // Write chunks with sizes
342 for (const auto& chunk : chunks) {
343 codecs::EliasGamma::encode(static_cast<uint32_t>(chunk.size()),
344 header_writer);
345 result.insert(result.end(), chunk.begin(), chunk.end());
346 }
347
348 return result;
349 }
